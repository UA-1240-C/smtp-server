158 void ThreadLoop(std::size_t id, const std::function<void(std::size_t)>& init, const std::stop_token &stop_tok)
159 {
160 try

312 template <typename Function>
313 void EnqueueTask(Function &&f)
314 {
316 auto i_opt = m_priority_queue.CopyFrontAndRotateToBack();
317 if (!i_opt.has_value()) return;
318
319 // get the index from the std::optional
320 auto i = *i_opt;
321
322 // increment the unassigned tasks and in flight tasks
323 m_unassigned_tasks.fetch_add(1, std::memory_order_release);
324
325 // reset the in flight signal if the list was previously empty
326 if (const auto prev_in_flight = m_in_flight_tasks.fetch_add(1, std::memory_order_release) == 0)
327 {
328 m_threads_complete_signal.store(false, std::memory_order_release);
329 }
330
331 // assign work
332 m_tasks[i].m_tasks.PushBack(std::forward<Function>(f));
333 m_tasks[i].m_signal.release();
334 }
